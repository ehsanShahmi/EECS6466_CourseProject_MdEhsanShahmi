    counter = collections.Counter()
    
    try:
        with open(csv_file, 'r') as f:
            reader = csv.reader(f)
            for row in reader:
                if row[0].startswith(emp_prefix):
                    counter[row[0]] += 1
    except FileNotFoundError:
        return {"error": f"The file {csv_file} was not found."}
    except Exception as e:
        return {"error": str(e)}
    
    return dict(counter)
