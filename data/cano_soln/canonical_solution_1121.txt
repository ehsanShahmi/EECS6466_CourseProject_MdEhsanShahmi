    urls = re.findall(r'(https?://[^\s,]+)', myString)
    geo_data = {}

    for url in urls:
        domain = urllib.parse.urlparse(url).netloc
        response = requests.get(f"http://ip-api.com/json/{domain}?access_key={API_KEY}")
        geo_data[domain] = json.loads(response.text)

    return geo_data
