    if not os.path.exists(directory):
        raise OSError("directory must exist.")

    counter = Counter()

    for suffix in extensions:
        count = len(
            glob.glob(os.path.join(directory, "**", "*" + suffix), recursive=True)
        )
        if count:
            counter[suffix] += count
        else:
            if keep_zero:
                counter[suffix] += count
    return counter
