    # Setting the random seed for reproducibility
    np.random.seed(seed)

    # Generating random points
    x = np.random.rand(n)
    y = np.random.rand(n)

    # Plotting
    fig, ax = plt.subplots()

    ax.set_title("Scatter plot of random points")
    ax.set_xlabel("X")
