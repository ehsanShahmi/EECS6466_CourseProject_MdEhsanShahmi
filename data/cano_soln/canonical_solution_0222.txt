    sorted_list = sorted(list_input, key=lambda x: (math.degrees(x), x))
    cumsum = np.cumsum(sorted_list)
    
    # Plotting the line chart
    ax = plt.plot(cumsum)[0].axes
    ax.set_title("Cumulative Sum Plot")
    ax.set_xlabel("Index")
