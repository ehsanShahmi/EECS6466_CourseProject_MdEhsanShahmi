    words = text.split()
    dollar_words = [
        word
        for word in words
        if word.startswith("$")
        and not all(c in PUNCTUATION for c in word)
        and len(word) > 1
    ]
    freq = nltk.FreqDist(dollar_words)
    if not freq:  # If frequency distribution is empty, return None
        return None
    plt.figure(figsize=(10, 5))
    sns.barplot(x=freq.keys(), y=freq.values())
    return plt.gca()
