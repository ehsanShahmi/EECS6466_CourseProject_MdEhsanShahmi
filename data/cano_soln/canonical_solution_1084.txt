    # Convert strings with commas to float, if applicable
    for col in df.columns:
        df[col] = pd.to_numeric(df[col].replace(",", "", regex=True), errors="coerce")

    std_devs = df.std()

    axes = []
    for col in df.columns:

        ax.set_title(col)
        axes.append(ax)
