    scaler = StandardScaler()

    fig, ax = plt.subplots()

    # Iterate over the datasets, scale each, and plot
    for i in range(len(x)):
        # Combine x and y values and scale them
        xy = np.vstack((x[i], y[i])).T  # Transpose to get correct shape for scaling
